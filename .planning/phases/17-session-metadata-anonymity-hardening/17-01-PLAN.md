<task type="auto">
  <name>Task 2: Normalize session metadata centrally</name>
  <files>src/server/session.ts</files>
  <read_first>
    - src/server/session.ts
    - src/types/index.ts
    - src/server/db/adapters/postgres.ts
  </read_first>
  <action>
    Update `src/server/session.ts`:

    - Import `createHmac` is already present; reuse it for metadata HMAC.
    - Import `SessionMetadataConfig` from `../types/index.js`.
    - Extend `SessionConfig` with `metadata?: SessionMetadataConfig`.
    - Add a local helper named `normalizeSessionMetadata`.

    Required behavior:

    ```ts
    const metadata = normalizeSessionMetadata({
      ipAddress: options.ipAddress,
      userAgent: options.userAgent,
      policy: config.metadata,
      secret: config.secret,
    });
    ```

    Then pass `metadata.ipAddress` and `metadata.userAgent` into `db.createSession`.

    Policy rules:
    - missing policy or missing individual field policy means `store`.
    - `omit` returns `undefined`.
    - `hash` returns `hmac-sha256:${hexDigest}` using `createHmac('sha256', secret).update(value).digest('hex')`.
    - IP `truncate` returns:
      - IPv4: `a.b.c.0/24` for valid dotted IPv4.
      - IPv6: first three non-empty hextets joined with `:` plus `::/48`; if parsing is ambiguous or invalid, return `undefined`.
      - invalid input: `undefined`, never raw fallback.
    - User-agent has no `truncate` mode.
  </action>
  <verify>
    <automated>rg -n "metadata\\?: SessionMetadataConfig|normalizeSessionMetadata|hmac-sha256|truncate" src/server/session.ts</automated>
  </verify>
  <acceptance_criteria>
    - `rg -c "normalizeSessionMetadata" src/server/session.ts` is at least `2` (definition plus call site).
    - `rg -c "hmac-sha256:" src/server/session.ts` is at least `1`.
    - `rg -n "ipAddress: metadata\\.ipAddress|userAgent: metadata\\.userAgent" src/server/session.ts` finds the `db.createSession` input.
  </acceptance_criteria>
</task>
